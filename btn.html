    <script>

      function copyURL() {

        const url = 'https://addinedu-rosanddl.oopy.io'; // 복사할 URL

        navigator.clipboard.writeText(url)

          .then(() => {

            alert("URL이 복사되었습니다: " + url);

          })

          .catch(err => {

            console.error("복사 실패:", err);

            alert("URL 복사에 실패했습니다.");

          });

      }

    </script>

  </head>
